215 Matrix<real,3,Dynamic> SolverVEM3D<real>::computeBPolygon(const shared_ptr<Polygon<3,real>>& polygon,MonomialsPolygon<real>& monomial) {
216  long numberOfPoints=polygon->numberOfPoints;
217  Matrix<real,3,Dynamic> BPolygon(3,numberOfPoints);
218 
219  // nella prima riga della matrice si usa P0, nel nostro caso 1/numberOfPoints
220  real firstLine=1.0/numberOfPoints;
221  for (int i=0;i<numberOfPoints;i++) {
222  BPolygon(0,i)=firstLine;
223  }
224 
225  // inizializzo il resto della matrice a 0
226  for (int i=1;i<3;i++) {
227  for (int j=0;j<numberOfPoints;j++) {
228  BPolygon(i,j)=0;
229  }
230  }
231 
232  // nelle altre uso un integrale al bordo
233 // auto& pointVector=polygon->pointVector;
234  for (int j=0;j<numberOfPoints;j++) { // j in questo caso è l'indice colonna della matrice
235  auto edge=(*polygon->point(j+1))-(*polygon->point(j));
236  auto edgeNormal=cross(edge,polygon->getNormal());
237  edgeNormal=edgeNormal/edgeNormal.norm(); // lo normalizzo
238  auto edgeLength=edge.norm();
239 
240 
241  real addTermX=monomial.gradient*edgeNormal[monomial.indexX]*edgeLength/2;
242  real addTermY=monomial.gradient*edgeNormal[monomial.indexY]*edgeLength/2;
243 
244  BPolygon(1,j)+=addTermX;
245  BPolygon(2,j)+=addTermY;
246  BPolygon(1,(j+1)%numberOfPoints)+=addTermX; // il % numberOfPoints per trattare il punto n+1
247  BPolygon(2,(j+1)%numberOfPoints)+=addTermY;
248  }
249 
250  return BPolygon;
251 }
