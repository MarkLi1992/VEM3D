30  long numberOfPoints=polygon->numberOfPoints;
31  Matrix<real,3,Dynamic> B(3,numberOfPoints);
32 
33  // prima riga
34  for (int j=0; j<numberOfPoints; j++) {
35  B(0,j)=1.0/numberOfPoints;
36  }
37 
38  // inizializzo il resto a 0
39  for (int i=1; i<3; i++) {
40  for (int j=0; j<numberOfPoints; j++) {
41  B(i,j)=0;
42  }
43  }
44 
45  // calcolo l'integrale al bordo per gli altri termini
46  real gradient=monomial.gradient;
47  for (int j=0; j<numberOfPoints; j++) {
48  // considero i due lati connessi al grado di libertà j
49  auto edge1=(*polygon->point(j))-(*polygon->point(j-1+numberOfPoints));
50  auto edge2=(*polygon->point(j+1))-(*polygon->point(j));
51 
52  Point<2,real> normalEdge1(edge1.y(),-edge1.x());
53  Point<2,real> normalEdge2(edge2.y(),-edge2.x());
54  // normalizzo
55  normalEdge1=normalEdge1/normalEdge1.norm();
56  normalEdge2=normalEdge2/normalEdge2.norm();
57 
58 // cout<<"point: "<<*pointVector[j]<<endl;

61

63  real edge2Norm=edge2.norm();
64 
65  B(1,j)=gradient*(normalEdge1.x()*edge1Norm+normalEdge2.x()*edge2Norm)/2;
66  B(2,j)=gradient*(normalEdge1.y()*edge1Norm+normalEdge2.y()*edge2Norm)/2;
67  }
